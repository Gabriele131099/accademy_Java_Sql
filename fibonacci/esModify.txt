    drop database fibonacciTest2;

    create database fibonacciTest2;

    use fibonacci2;
    create table user(
            id primary key unique
            username varchar(30) unique,
            password varchar(30),
            nazionalita varchar(30),
            annoNascita varchar(30),
            mail varchar(50)
    );

    create table Statistiche(
            id pk, unique
            idUser int,
            numeroLogin int,
            dateUltimoLogin varchar(30),
            numeroPreferito int,
            numeroDiVolteGenerato int,
            numeroDiVolteCercato int,
            numeroPiuAltoGenerato int,
            numeroPiuAltoCercato int,
            FOREIGN KEY (idUser) REFERENCES user(id)
    );
